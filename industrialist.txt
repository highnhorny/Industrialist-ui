local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")

local MAX_POLLUTION = 50000000
local UPDATE_INTERVAL = 1 

local screenGui = Instance.new("ScreenGui")
screenGui.Name = "PollutionDisplayGui"
screenGui.Parent = playerGui
screenGui.ResetOnSpawn = false
screenGui.IgnoreGuiInset = true

local mainFrame = Instance.new("Frame")
mainFrame.Name = "MainFrame"
mainFrame.Parent = screenGui
mainFrame.BackgroundColor3 = Color3.fromRGB(20, 20, 25)
mainFrame.BackgroundTransparency = 0.2
mainFrame.BorderSizePixel = 0
mainFrame.Position = UDim2.new(0.5, -150, 0, 50)
mainFrame.Size = UDim2.new(0, 300, 0, 150)

local mainCorner = Instance.new("UICorner")
mainCorner.CornerRadius = UDim.new(0, 10)
mainCorner.Parent = mainFrame

local mainStroke = Instance.new("UIStroke")
mainStroke.Thickness = 2
mainStroke.Color = Color3.fromRGB(60, 60, 70)
mainStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
mainStroke.Parent = mainFrame

local titleLabel = Instance.new("TextLabel")
titleLabel.Name = "TitleLabel"
titleLabel.Parent = mainFrame
titleLabel.BackgroundTransparency = 1
titleLabel.Position = UDim2.new(0, 15, 0, 5)
titleLabel.Size = UDim2.new(1, -90, 0, 30)
titleLabel.Font = Enum.Font.GothamBold
titleLabel.Text = "ENVIRONMENTAL STATS"
titleLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
titleLabel.TextSize = 12
titleLabel.TextXAlignment = Enum.TextXAlignment.Left

local buttonContainer = Instance.new("Frame")
buttonContainer.Size = UDim2.new(0, 60, 0, 30)
buttonContainer.Position = UDim2.new(1, -65, 0, 5)
buttonContainer.BackgroundTransparency = 1
buttonContainer.Parent = mainFrame

local listLayout = Instance.new("UIListLayout")
listLayout.FillDirection = Enum.FillDirection.Horizontal

local function createHeaderButton(name, text, color)
    local btn = Instance.new("TextButton")
    btn.Name = name
    btn.Size = UDim2.new(0, 24, 0, 24)
    btn.BackgroundColor3 = color
    btn.Font = Enum.Font.GothamBold
    btn.Text = text
    btn.TextColor3 = Color3.fromRGB(255, 255, 255)
    btn.TextSize = 14
    btn.Parent = buttonContainer
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 6)
    return btn
end

local unloadButton = createHeaderButton("UnloadBtn", "⏻", Color3.fromRGB(80, 80, 90))
local closeButton = createHeaderButton("CloseBtn", "X", Color3.fromRGB(200, 50, 50))

local pollutionHistory = {}
local moneyHistory = {}
local lastUpdate = 0
local connection

local function unload()
    if connection then connection:Disconnect() end
    screenGui:Destroy()
end

unloadButton.MouseButton1Click:Connect(unload)
closeButton.MouseButton1Click:Connect(function()
    mainFrame.Visible = false
end)

local function createStatLabel(name, pos, size, color, fontSize)
    local lbl = Instance.new("TextLabel")
    lbl.Name = name
    lbl.Parent = mainFrame
    lbl.BackgroundTransparency = 1
    lbl.Position = pos
    lbl.Size = size
    lbl.Font = Enum.Font.GothamMedium
    lbl.Text = "..."
    lbl.TextColor3 = color
    lbl.TextSize = fontSize
    lbl.TextXAlignment = Enum.TextXAlignment.Center
    return lbl
end

local pollutionLabel = createStatLabel("PollutionLabel", UDim2.new(0, 10, 0, 40), UDim2.new(0.5, -15, 0, 40), Color3.fromRGB(255, 100, 100), 22)
local increaseLabel = createStatLabel("IncreaseLabel", UDim2.new(0.5, 5, 0, 40), UDim2.new(0.5, -15, 0, 40), Color3.fromRGB(255, 150, 150), 16)
local moneyPerHourLabel = createStatLabel("MoneyPerHour", UDim2.new(0, 10, 0, 90), UDim2.new(1, -20, 0, 25), Color3.fromRGB(100, 255, 150), 18)
local moneyPerSecLabel = createStatLabel("MoneyPerSec", UDim2.new(0, 10, 0, 115), UDim2.new(1, -20, 0, 20), Color3.fromRGB(150, 255, 255), 14)

local dragging, dragStart, startPos
mainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = mainFrame.Position
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        mainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

mainFrame.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end
end)

local function updateStats()
    local pollutionValue = ReplicatedStorage:FindFirstChild("Pollution")
    local moneyValue = ReplicatedStorage:FindFirstChild("Money")
    local now = tick()
    
    if pollutionValue and pollutionValue:IsA("NumberValue") then
        local val = pollutionValue.Value
        table.insert(pollutionHistory, {t = now, v = val})
        pollutionLabel.Text = string.format("%.2f%%", (val / MAX_POLLUTION) * 100)
        
        while #pollutionHistory > 0 and now - pollutionHistory[1].t > 3600 do table.remove(pollutionHistory, 1) end
        
        if #pollutionHistory > 1 then
            local elapsed = now - pollutionHistory[1].t
            local diff = val - pollutionHistory[1].v
            local hourlyRate = (diff / elapsed) * 3600
            increaseLabel.Text = string.format("+%.2f%%/hr", (hourlyRate / MAX_POLLUTION) * 100)
        end
    end
    
    if moneyValue and moneyValue:IsA("NumberValue") then
        local val = moneyValue.Value
        table.insert(moneyHistory, {t = now, v = val})
        while #moneyHistory > 0 and now - moneyHistory[1].t > 3600 do table.remove(moneyHistory, 1) end
        
        if #moneyHistory > 1 then
            local totalGained = 0
            for i = 2, #moneyHistory do
                local gain = moneyHistory[i].v - moneyHistory[i-1].v
                if gain > 0 then totalGained += gain end
            end
            local elapsed = now - moneyHistory[1].t
            local mPerHour = (totalGained / elapsed) * 3600
            moneyPerHourLabel.Text = string.format("Income: +$%s/hr", math.floor(mPerHour))
            moneyPerSecLabel.Text = string.format("$%.2f per second", mPerHour / 3600)
        end
    end
end

connection = RunService.Heartbeat:Connect(function()
    if tick() - lastUpdate >= UPDATE_INTERVAL then
        lastUpdate = tick()
        updateStats()
    end
end)
