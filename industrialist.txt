local function createStatLabel(name, pos, size, color, fontSize, align)
	local lbl = Instance.new("TextLabel")
	lbl.Name = name
	lbl.Parent = mainFrame
	lbl.BackgroundTransparency = 1
	lbl.Position = pos
	lbl.Size = size
	lbl.Font = Enum.Font.GothamMedium
	lbl.Text = "..."
	lbl.TextColor3 = color
	lbl.TextSize = fontSize
	lbl.TextXAlignment = align or Enum.TextXAlignment.Center
	return lbl
end

local increaseLabel = createStatLabel("IncreaseLabel", UDim2.new(0.5, 5, 0, 50), UDim2.new(0.5, -20, 0, 30), Color3.fromRGB(255, 255, 255), 14, Enum.TextXAlignment.Right)

local moneyPerHourLabel = createStatLabel("MoneyPerHour", UDim2.new(0, 0, 0, 100), UDim2.new(1, 0, 0, 25), Color3.fromRGB(100, 255, 150), 18, Enum.TextXAlignment.Center)
local moneyPerSecLabel = createStatLabel("MoneyPerSec", UDim2.new(0, 0, 0, 125), UDim2.new(1, 0, 0, 20), Color3.fromRGB(150, 255, 255), 14, Enum.TextXAlignment.Center)

local pollutionHistory = {}
local moneyHistory = {}
local lastUpdate = 0
local connection

local function unload()
	if connection then connection:Disconnect() end
	screenGui:Destroy()
end

closeButton.MouseButton1Click:Connect(unload)

UserInputService.InputBegan:Connect(function(input, gp)
	if not gp and input.KeyCode == MINIMIZE_KEY then
		mainFrame.Visible = not mainFrame.Visible
	end
end)

local dragging, dragStart, startPos
mainFrame.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = mainFrame.Position
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
		local delta = input.Position - dragStart
		mainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
end)

mainFrame.InputEnded:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then dragging = false end
end)

local function formatNumber(n)
	if n >= 1000000 then return string.format("%.2fM", n / 1000000)
	elseif n >= 1000 then return string.format("%.1fK", n / 1000) end
	return tostring(math.floor(n))
end

local function updateStats()
	local pollutionValue = ReplicatedStorage:FindFirstChild("Pollution")
	local moneyValue = ReplicatedStorage:FindFirstChild("Money")
	local now = tick()
	
	if pollutionValue and pollutionValue:IsA("NumberValue") then
		local val = pollutionValue.Value
		local percent = (val / MAX_POLLUTION)
		table.insert(pollutionHistory, {t = now, v = val})
		
		pollutionLabel.Text = string.format("%.2f%%", math.clamp(percent * 100, 0, 100))
		pollutionLabel.TextColor3 = Color3.fromRGB(255, 255, 255):Lerp(Color3.fromRGB(255, 50, 50), percent)
		
		while #pollutionHistory > 0 and now - pollutionHistory[1].t > 3600 do table.remove(pollutionHistory, 1) end
		
		if #pollutionHistory > 1 then
			local elapsed = now - pollutionHistory[1].t
			local diff = val - pollutionHistory[1].v
			local hourlyRate = (diff / elapsed) * 3600
			local increasePercent = (hourlyRate / MAX_POLLUTION) * 100
			
			if math.abs(increasePercent) < 0.001 then
				increaseLabel.Text = "0.00%/hr"
				increaseLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
			elseif increasePercent > 0 then
				increaseLabel.Text = string.format("+%.2f%%/hr", increasePercent)
				increaseLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
			else
				increaseLabel.Text = string.format("%.2f%%/hr", increasePercent)
				increaseLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
			end
		end
	end
	
	if moneyValue and moneyValue:IsA("NumberValue") then
		local val = moneyValue.Value
		table.insert(moneyHistory, {t = now, v = val})
		while #moneyHistory > 0 and now - moneyHistory[1].t > 3600 do table.remove(moneyHistory, 1) end
		
		if #moneyHistory > 1 then
			local totalGained = 0
			for i = 2, #moneyHistory do
				local gain = moneyHistory[i].v - moneyHistory[i-1].v
				if gain > 0 then totalGained += gain end
			end
			local elapsed = now - moneyHistory[1].t
			local mPerHour = (totalGained / elapsed) * 3600
			moneyPerHourLabel.Text = "+$" .. formatNumber(mPerHour) .. " / hour"
			moneyPerSecLabel.Text = "$" .. string.format("%.2f", mPerHour / 3600) .. " / sec"
		end
	end
end

connection = RunService.Heartbeat:Connect(function()
	if tick() - lastUpdate >= UPDATE_INTERVAL then
		lastUpdate = tick()
		updateStats()
	end
end)
